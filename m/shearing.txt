#include <GL/glut.h>
using namespace std;

void drawLineGL(int x0, int y0, int x1, int y1) {
    glBegin(GL_LINES);
    glVertex2i(x0, y0);
    glVertex2i(x1, y1);
    glEnd();
}
void drawTriangle(int x1, int y1, int x2, int y2, int x3, int y3) {
    drawLineGL(x1, y1, x2, y2);
    drawLineGL(x2, y2, x3, y3);
    drawLineGL(x3, y3, x1, y1);
}
void shear(int &x, int &y, float shx, float shy, int cx, int cy) {
    int dx = x - cx;
    int dy = y - cy;
    int newX = dx + shx * dy;
    int newY = dy + shy * dx;
    x = cx + newX;
    y = cy + newY;
}
void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    int x1 = 100, y1 = 100, x2 = 200, y2 = 100, x3 = 150, y3 = 200;
    glColor3f(1, 1, 1);
    drawTriangle(x1, y1, x2, y2, x3, y3);

    int cx = (x1 + x2 + x3) / 3;
    int cy = (y1 + y2 + y3) / 3;
    shear(x1, y1, 0.5, 0.0, cx, cy);
    shear(x2, y2, 0.5, 0.0, cx, cy);
    shear(x3, y3, 0.5, 0.0, cx, cy);
    glColor3f(0, 1, 1);
    drawTriangle(x1, y1, x2, y2, x3, y3);
    glFlush();
}
void init() {
    glClearColor(0, 0, 0, 1);
    gluOrtho2D(0, 500, 0, 500);
}
int main(int argc, char **argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Shearing - Moinuddin Ahmed");
    init();
    glutDisplayFunc(display);
    glutMainLoop();
    return 0;
}
