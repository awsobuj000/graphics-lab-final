#include <GL/glut.h>
#include <math.h>
float x1 = 10, Y1 = 10;
float x2 = 70, y2 = 10;
float x3 = 10, y3 = 40;
float x4 = 70, y4 = 40;
void drawImage(float a1, float b1,float a2, float b2, float a3, float b3, float a4, float b4)
{
glLineWidth(2.0);
glBegin(GL_LINES);
glVertex2f(a1, b1);
glVertex2f(a2, b2);
glVertex2f(a2, b2);
glVertex2f(a4, b4);
glVertex2f(a3, b3);
glVertex2f(a4, b4);
glVertex2f(a1, b1);
glVertex2f(a3, b3);
glEnd();
}
void display()
{
glClear(GL_COLOR_BUFFER_BIT);
glColor3f(0, 0, 0);
drawImage(x1, Y1, x2, y2, x3, y3, x4, y4);
glFlush();
}
void keyboard(unsigned char key, int x, int y)
{
glClear(GL_COLOR_BUFFER_BIT);
glColor3f(0, 0, 0);
drawImage(x1, Y1, x2, y2, x3, y3, x4, y4);
switch (key)
{
case 't':
{
glColor3f(1,0,0);
float tx = 50, ty = 20;
drawImage(x1+tx, Y1+ty, x2+tx, y2+ty, x3+tx, y3+ty, x4+tx, y4+ty);
break;
}
case 's':
{
glColor3f(1,0,0);
float sx = 1.3, sy = 1.4;
drawImage(x1*sx, Y1*sy, x2*sx, y2*sy, x3*sx, y3*sy, x4*sx, y4*sy);
break;
}
case 'r':
{
glColor3f(1,0,0);
float angle = 30 * 3.1416 / 180;
drawImage(x1 * cos(angle) - Y1 * sin(angle),
x1 * sin(angle) + Y1 * cos(angle),
x2 * cos(angle) - y2 * sin(angle),
x2 * sin(angle) + y2 * cos(angle),
x3 * cos(angle) - y3 * sin(angle),
x3 * sin(angle) + y3 * cos(angle),
x4 * cos(angle) - y4 * sin(angle),
x4 * sin(angle) + y4 * cos(angle));
break;
}
case 'f':
{
glColor3f(1,0,0);
drawImage(x1, -Y1, x2, -y2, x3, -y3, x4, -y4);
drawImage(-x1, Y1, -x2, y2, -x3, y3, -x4, y4);
break;
}
case 'h':
{
glColor3f(1,0,0);
float sh = 0.5;
drawImage(x1, Y1, x2+sh*y2, y2, x3+sh*y3, y3, x4+sh*y4, y4);
drawImage(x1, Y1, x2, y2+sh*x2, x3, y3+sh*x3, x4, y4+sh*x4);
break;
}
}
glFlush();
}
void init()
{
glClearColor(1, 1, 1, 1);
gluOrtho2D(-150, 150, -150, 150);
}
int main(int argc, char** argv)
{
glutInit(&argc, argv);
glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
glutInitWindowSize(500, 500);
glutCreateWindow("2D Transformations");
init();
glutDisplayFunc(display);
glutKeyboardFunc(keyboard);
glutMainLoop();
return 0;
}
