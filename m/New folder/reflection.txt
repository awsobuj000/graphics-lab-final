#include <GL/glut.h>
#include <iostream>
using namespace std;

// User-input triangle coordinates
int x1_, y1_, x2_, y2_, x3_, y3_;

void drawLineGL(int x0, int y0, int x1, int y1) {
    glBegin(GL_LINES);
    glVertex2i(x0, y0);
    glVertex2i(x1, y1);
    glEnd();
}

void drawTriangle(int a1, int b1, int a2, int b2, int a3, int b3) {
    drawLineGL(a1, b1, a2, b2);
    drawLineGL(a2, b2, a3, b3);
    drawLineGL(a3, b3, a1, b1);
}

void reflectXY(int &x, int &y, int cx, int cy) {
    x = cx - (x - cx);   // Reflect across centroid X
    y = cy - (y - cy);   // Reflect across centroid Y
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    // Draw original triangle (white)
    glColor3f(1, 1, 1);
    drawTriangle(x1_, y1_, x2_, y2_, x3_, y3_);

    // Centroid
    int cx = (x1_ + x2_ + x3_) / 3;
    int cy = (y1_ + y2_ + y3_) / 3;

    // Copy points for reflection
    int r1x = x1_, r1y = y1_;
    int r2x = x2_, r2y = y2_;
    int r3x = x3_, r3y = y3_;

    reflectXY(r1x, r1y, cx, cy);
    reflectXY(r2x, r2y, cx, cy);
    reflectXY(r3x, r3y, cx, cy);

    // Draw reflected triangle (magenta)
    glColor3f(1, 0, 1);
    drawTriangle(r1x, r1y, r2x, r2y, r3x, r3y);

    glFlush();
}

void init() {
    glClearColor(0, 0, 0, 1);
    gluOrtho2D(0, 500, 0, 500);
}

int main(int argc, char **argv) {

    cout << "Enter x1 y1: ";
    cin >> x1_ >> y1_;

    cout << "Enter x2 y2: ";
    cin >> x2_ >> y2_;

    cout << "Enter x3 y3: ";
    cin >> x3_ >> y3_;

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500, 500);
    glutCreateWindow("Reflection - User Input");

    init();
    glutDisplayFunc(display);
    glutMainLoop();
}
