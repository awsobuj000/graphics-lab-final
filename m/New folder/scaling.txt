#include <GL/glut.h>
#include <iostream>
using namespace std;

int x1_, y1_, x2_, y2_, x3_, y3_;

void drawLineGL(int a, int b, int c, int d) {
    glBegin(GL_LINES);
    glVertex2i(a,b);
    glVertex2i(c,d);
    glEnd();
}

void drawTriangle(int a1, int b1, int a2, int b2, int a3, int b3) {
    drawLineGL(a1,b1,a2,b2);
    drawLineGL(a2,b2,a3,b3);
    drawLineGL(a3,b3,a1,b1);
}

void scale(int &x, int &y, float sx, float sy, int cx, int cy) {
    x = cx + (x - cx) * sx;
    y = cy + (y - cy) * sy;
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    glColor3f(1,1,1);
    drawTriangle(x1_, y1_, x2_, y2_, x3_, y3_);

    int cx = (x1_ + x2_ + x3_) / 3;
    int cy = (y1_ + y2_ + y3_) / 3;

    int sx1=x1_, sy1=y1_, sx2=x2_, sy2=y2_, sx3=x3_, sy3=y3_;
    scale(sx1, sy1, 1.5, 1.5, cx, cy);
    scale(sx2, sy2, 1.5, 1.5, cx, cy);
    scale(sx3, sy3, 1.5, 1.5, cx, cy);

    glColor3f(0,1,0);
    drawTriangle(sx1, sy1, sx2, sy2, sx3, sy3);

    glFlush();
}

void init() {
    glClearColor(0,0,0,1);
    gluOrtho2D(0,500,0,500);
}

int main(int argc, char **argv) {
    cout << "Enter x1 y1: "; cin >> x1_ >> y1_;
    cout << "Enter x2 y2: "; cin >> x2_ >> y2_;
    cout << "Enter x3 y3: "; cin >> x3_ >> y3_;

    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(500,500);
    glutCreateWindow("Scaling - User Input");

    init();
    glutDisplayFunc(display);
    glutMainLoop();
}
